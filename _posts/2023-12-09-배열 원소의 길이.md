```java
import java.util.*;

class Solution {
    public int[] solution(String[] strlist) {
        int[] answer = new int[strlist.length];
        
        for (int i = 0; i < strlist.length; i++) {
            answer[i] = getLength(strlist[i]);
        }
        
        return answer;
    }
    
    private int getLength(String str) {
        return str.length();
    }
}
```
